B) Add components

/components/JsonLd.tsx

import Head from "next/head";
import ecosys from "@/seo/ecosystem.json";

export default function JsonLd() {
  const data = {
    "@context": "https://schema.org",
    "@type": "Organization",
    "name": ecosys.organization.name,
    "url": ecosys.organization.url,
    "logo": ecosys.organization.logo,
    "sameAs": ecosys.organization.sameAs,
    "brand": ecosys.products.map(p => ({
      "@type": "Brand",
      "name": p.name,
      "url": p.url
    }))
  };
  return (
    <Head>
      <script type="application/ld+json" dangerouslySetInnerHTML={{ __html: JSON.stringify(data) }} />
    </Head>
  );
}


/components/ProductGrid.tsx

import Link from "next/link";
import ecosys from "@/seo/ecosystem.json";

export default function ProductGrid() {
  return (
    <div className="grid sm:grid-cols-2 gap-6">
      {ecosys.products.map(p => (
        <div key={p.url} className="rounded-xl p-6 border border-white/10">
          <h3 className="text-xl font-semibold">{p.name}</h3>
          <p className="opacity-80">{p.tagline}</p>
          <Link href={p.url} target="_blank" rel="noopener noreferrer" className="mt-3 inline-block underline">
            {p.anchors[0]}
          </Link>
        </div>
      ))}
    </div>
  );
}


/components/EcosystemFooter.tsx

import Link from "next/link";
import ecosys from "@/seo/ecosystem.json";

export default function EcosystemFooter() {
  return (
    <footer aria-label="Ecosystem" className="border-t border-white/10 mt-16 pt-10">
      <div className="grid gap-4 sm:grid-cols-3">
        <div>
          <p className="text-sm opacity-75">{ecosys.organization.name}</p>
          <ul className="mt-2 space-y-1">
            {ecosys.products.map(p => (
              <li key={p.url}>
                <Link href={p.url} target="_blank" rel="noopener noreferrer" className="underline hover:no-underline">
                  {p.name} – {p.tagline}
                </Link>
              </li>
            ))}
          </ul>
        </div>
        <div>
          <p className="text-sm opacity-75">Products</p>
          <ul className="mt-2 space-y-1">
            {ecosys.products.map(p => (
              <li key={p.name}>
                <Link href={p.url} target="_blank" rel="noopener noreferrer">
                  {p.anchors?.[0] ?? p.name}
                </Link>
              </li>
            ))}
          </ul>
        </div>
        <div>
          <p className="text-sm opacity-75">Company</p>
          <ul className="mt-2 space-y-1">
            <li>
              <Link href={ecosys.organization.url} className="underline" target="_blank" rel="noopener noreferrer">
                Automate AI
              </Link>
            </li>
          </ul>
        </div>
      </div>
      <p className="mt-8 text-xs opacity-60">© {new Date().getFullYear()} {ecosys.organization.name}</p>
    </footer>
  );
}

C) Wire into your layout

In app/layout.tsx (or _app.tsx if using pages router):

import "./globals.css";
import JsonLd from "@/components/JsonLd";
import EcosystemFooter from "@/components/EcosystemFooter";

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <head>
        <JsonLd />
      </head>
      <body>
        {children}
        <EcosystemFooter />
      </body>
    </html>
  );
}
